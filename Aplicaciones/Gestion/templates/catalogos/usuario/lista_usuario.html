def reiniciar_password(request): """ Genera una nueva contraseña aleatoria para un usuario y la envía por correo. """ if request.method != 'POST': messages.error(request, "Método no permitido") return redirect('/listadousuarios/') try: usuario_id = request.POST.get('usuario_id') if not usuario_id: messages.error(request, "ID de usuario no proporcionado") return redirect('/listadousuarios/') usuario = Usuario.objects.get(id_us=usuario_id) # Generar contraseña aleatoria de 10 caracteres import string import random nueva_password = ''.join(random.choices( string.ascii_letters + string.digits + "!@#$%^&*", k=10 )) # Hashear y guardar usuario.password_us = make_password(nueva_password) usuario.intentos_fallidos_us = 0 usuario.bloqueado_hasta_us = None usuario.save() # Enviar correo con la nueva contraseña try: send_mail( subject='🔐 Nueva contraseña - Hacienda Joseph', message=f''' Hola {usuario.nombre_us}, Se ha reiniciado tu contraseña en el Sistema de Gestión Ganadera Hacienda Joseph. Tu nueva contraseña es: {nueva_password} Te recomendamos cambiarla después de iniciar sesión. Si no solicitaste este cambio, contacta al administrador inmediatamente. Saludos, Equipo Hacienda Joseph ''', from_email=settings.DEFAULT_FROM_EMAIL, recipient_list=[usuario.email_us], fail_silently=False, ) messages.success(request, f"Contraseña reiniciada exitosamente. Nueva contraseña enviada a {usuario.email_us}") except Exception as e: messages.warning(request, f"Contraseña reiniciada, pero no se pudo enviar el correo: {str(e)}") except Usuario.DoesNotExist: messages.error(request, "Usuario no encontrado") except Exception as e: messages.error(request, f"Error al reiniciar contraseña: {str(e)}") return redirect('/listadousuarios/')
